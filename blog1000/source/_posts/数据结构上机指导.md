1.1 求素数
```c++
#include<stdio.h>
#include<math.h>
#include<iostream>

bool prime(int n)
{
	int i;
	for (i = 2; i <= (int)sqrt(n); i++) {
		if (n%i == 0)
			return false;
		else
			return true;
	}
}
int main()
{
	int n, i, j = 0;
	printf("n:" );
	scanf("%d",&n);
	printf("小于等于%d的素数", n);
	if (n>2)
	{
		printf("%4d", n);
		j++;
	}
	for (i = 3; i <= n; i += 2)
		if (prime(i))
		{
			printf("%4d", i);
			if (j != 0 && ++j % 10 == 0)
				printf("\n");
		}
	printf("\n");
}
```

```c++
#include<stdio.h>
#include<string.h>
#define MAX 100
//算法
bool func(char s[])
{
  bool flag=true;
  int i,j,slen=strlen(s);
  for(i=0,j=slen-1;i<j;i++,j--){
    if(s[i]!=s[j]){
      flag=false;
      break;
    }
  }
  return (flag);
}

int main()
{
```
